gitlab(docker)+git免密+CICD

# 安装gitlab

docker pull gitlab/gitlab-ce
mkdir -p /docker/gitlab/etc
mkdir -p /docker/gitlab/log
mkdir -p /docker/gitlab/opt
chmod -R 777 /docker/gitlab
cd /docker/gitlab/
vim docker-compose.yml

version: '3'

services:
  gitlabce:
    image: 'gitlab/gitlab-ce:latest'
    container_name: gitlabce
    hostname: gitlabce
    restart: always
    volumes:
      - /docker/gitlab/etc:/etc/gitlab
      - /docker/gitlab/log:/var/log/gitlab
      - /docker/gitlab/opt:/var/opt/gitlab
    ports:
      - '9980:80'
      - '9922:22'

docker-compose up -d

可能会花几分钟，监听一下,成功启动后开始下一步

watch docker ps

vim /docker/gitlab/etc/gitlab.rb

external_url 'http://192.168.2.200' # 不要加端口
gitlab_rails['gitlab_ssh_host'] = '192.168.2.200'
gitlab_rails['gitlab_shell_ssh_port'] = 9922

docker exec -it gitlabce /bin/bash
gitlab-ctl reconfigure
exit

vim /docker/gitlab/opt/gitlab-rails/etc/gitlab.yml

host: 192.168.2.200
port: 9980

docker exec -it gitlabce /bin/bash
gitlab-ctl restart
exit

# 进入容器修改密码

docker exec -it gitlabce /bin/bash
gitlab-rails console

比较耗时,等待完成后依次输入以下语句

u=User.where(id:1).first
u.password='614489858' // 新密码
u.password_confirmation='614489858'
u.save!
exit

浏览器访问http://192.168.2.200:9980/,账号是root，密码是刚才修改的密码,也就是614489858.

# 安装gitlab-runner(CI/CD)

docker pull gitlab/gitlab-runner
mkdir -p /docker/gitlab-runner/config
cd  /docker/gitlab-runner
vim docker-compose.yml

version: '3'

services:
  gitlabrunner:
    image: 'gitlab/gitlab-runner:latest'
    container_name: gitlabrunner
    hostname: gitlabrunner
    restart: always
    volumes:
      - /docker/gitlab-runner/config:/etc/gitlab-runner
      - /var/run/docker.sock:/var/run/docker.sock

docker-compose up -d

# 注册gitlab-runner

root登录gitlab,左上角点击menu按钮 -> 选择Admin -> 点击左侧菜单Runners -> 点击右上方按钮Register an instance runner获取token

zgAq_h8xeeq8xbtSMsBg

回到虚拟机

docker run --rm \
-v /docker/gitlab-runner/config:/etc/gitlab-runner \
gitlab/gitlab-runner register \
--non-interactive \
--executor "docker" \
--docker-image alpine:latest \
--url "http://192.168.2.200:9980/" \
--registration-token "zgAq_h8xeeq8xbtSMsBg" \
--description "runner " \
--tag-list "runner" \
--run-untagged="true" \
--locked="false"

参数说明

  • --registration-token是刚才在gitlab获取的token
  • --url是gitlab地址(这里是http://192.168.2.200:9980/)

注册成功后刷新刚才页面可以看到

# 远程部署

主要利用远程ssh

机器A生成一对密钥，公钥传给机器B

机器A操作

// 添加host，免得每次写一大串
// stu是机器A， s1是机器B
vim /etc/hosts
192.168.2.200 stu
192.168.2.201 s1

// 同步到机器B
scp /etc/hosts root@s1:/etc/hosts

// 生成密钥
ssh-keygen -t rsa -C "stu"

// 需要改名
scp ~/.ssh/id_rsa.pub root@s1:~/.ssh/authorized_keys
// 机器B报错没有.ssh/文件夹则创建一个则可

// 测试
ssh s1

# job_install:
#   stage: install
#   tags:
#     - "runner"
#   script:
#     # - npm install
#     - echo ${containerName}
#     - echo ${imageName}

job_build:
  stage: build
  image: docker
  script:
    - docker build -t ${imageName} .
  only: 
    - master
  interruptible: true # 可被打断，需要gitlab设置，参见下面注释

job_deploy:
  stage: deploy
  image: docker
  before_script:
    - 'which ssh-agent || ( yum update -y && yum install openssh-client git -y )'
    - eval $(ssh-agent -s)
    - echo "$SSH_PRIVATE_KEY" | tr -d '\r' | ssh-add -
    - mkdir -p ~/.ssh
    - chmod 700 ~/.ssh
    - ssh-keyscan ${target} >> ~/.ssh/known_hosts 
    - chmod 644 ~/.ssh/known_hosts
  script:
    - docker save -o ${imageName}.tar ${imageName} 
    - scp ./${imageName}.tar root@${target}:/root 
    - echo ${containerName}
    - echo ${imageName}
    - ssh ${target} "if docker ps -aq --filter name=${containerName}; then docker stop ${containerName} && docker rm ${containerName} && docker rmi ${imageName};fi"
    - ssh ${target} "docker load -i ${imageName}.tar"
    - ssh ${target} "docker run -d -p 3000:3000 --name ${containerName} ${imageName}"
    - ssh ${target} "rm -rf ./${imageName}.tar"
  timeout: 3h 30m # 超时时间
  # when: manual
