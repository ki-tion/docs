# 过滤器

vim /src/filter/execption.ts

import { ArgumentsHost, Catch, ExceptionFilter, HttpException } from "@nestjs/common"

@Catch(HttpException)
export class MyHttpExecption implements ExceptionFilter<HttpException> {
    catch(exception: HttpException, host: ArgumentsHost) {
        const ctx = host.switchToHttp();
        const res = ctx.getResponse();
        const req = ctx.getRequest();
        const status = exception.getStatus();
        const execptionRes = exception.getResponse() as any;
        const {error, message} = execptionRes
        res.status(status).json({
            status,
            timestamp: Date.now(),
            path: req.url,
            error,
            message
        })
    }
}

# 守卫（权限验证）

vim /src/role-guard/role.ts

import { CanActivate, ExecutionContext, Injectable } from "@nestjs/common"
import { Reflector } from "@nestjs/core"
import { Observable } from "rxjs"

@Injectable()
export class RoleGuard implements CanActivate {
    constructor(private readonly reflector: Reflector) {}
    canActivate(context: ExecutionContext): boolean | Promise<boolean> | Observable<boolean> {
        console.log("enter")
        const roles = this.reflector.get<string[]>("roles", context.getHandler())
        if(!roles) {
            return true
        }
        const req = context.switchToHttp().getRequest() 
        const { role } = req.query // 实际开发因该从hearders中取token，这里为了演示简单
        return !!roles.find(ele => ele === role)
    }
}

vim /src/role-guard/roleDecorator.ts (装饰器)

import { SetMetadata } from "@nestjs/common"
export const Roles = (...roles: string[]) => SetMetadata("roles", roles)

# 集成安装过滤器、守卫、api文档和安全设置

vim /src/main.ts

import { NestFactory, Reflector } from '@nestjs/core';
import { AppModule } from './app.module';
import { ValidationPipe } from '@nestjs/common';
// 生成api文档依赖
import { DocumentBuilder, SwaggerModule } from "@nestjs/swagger";
// hearder安全依赖
import helmet from 'helmet';
// 限制请求数量
import rateLimit from 'express-rate-limit';
// import cookieParser from "cookie-parser"
// import csurf from "csurf"
import { MyHttpExecption } from "./filter/execption"
import { RoleGuard } from "./role-guard/role"

(async () => {
  const app = await NestFactory.create(AppModule, {
    cors: true, // 设置跨站访问
  });

  // 使用跨站脚本攻击类的库
  app.use(helmet())
  // app.use(cookieParser())
  // app.use(csurf({ cookie: true }))
  app.use(
    rateLimit({
      windowMs: 15 * 60 * 1000, // 15 minutes
      max: 100, // 限制每个窗口最多 100 个请求
    }),
  )
  // 全局使用管道(验证数据类型)
  app.useGlobalPipes(new ValidationPipe({
    transform: true
  }));
  // 全局过滤器
  app.useGlobalFilters(new MyHttpExecption())
  // 全局开启权限
  app.useGlobalGuards(new RoleGuard(app.get(Reflector)))

  // 配置api 文档
  const swaggerOptions = new DocumentBuilder()
    .setTitle("nest api document")
    .setDescription("api 文档")
    .setVersion("1.0")
    .addBearerAuth()
    .build();
  const document = SwaggerModule.createDocument(app, swaggerOptions);
  // localhost:3000/doc
  SwaggerModule.setup("doc", app, document);
  
  await app.listen(3000);
})();

import { Entity, PrimaryGeneratedColumn, Column, Timestamp } from 'typeorm';
import { IsNotEmpty, IsString, Length, IsInt, Min, Max, IsOptional } from "class-validator"
import { ApiProperty } from "@nestjs/swagger"

@Entity()
export class Cat {
  // 会以类名来创建表,如果是驼峰命名的,生成的表名是下划线区分
  @PrimaryGeneratedColumn({ comment: '主键id' })
  @IsNotEmpty({message: "id不能为空"})
  @IsInt({message: "int类型"})
  @ApiProperty({description: "id", example: 1, required: true, nullable: false})
  id: number;

  @Column({ type: 'varchar', length: 10, comment: '名字', default: "" })
  @IsNotEmpty({message: "名称不能为空"})
  @IsString({message: "字符串类型"})
  @Length(2,10, {message: "2-10个字符"})
  @ApiProperty({description: "名称", example: "kition", required: true, nullable: false})
  name: string;
  
  @Column({ type: "int", comment: '年龄', default: 0})
  @IsNotEmpty({message: "年龄不能为空"})
  @IsInt({message: "int类型"})
  @Min(0, {message: "最小0"})
  @Max(100, {message: "最大100"})
  @ApiProperty({description: "年龄", example: 20, required: true, nullable: false})
  age: number;
  
  @Column({
    type: 'varchar',
    length: 30,
    comment: '颜色',
    // nullable: true,
    default: "",
  })
  @IsOptional() // 可选
  @IsNotEmpty({message: "颜色不能为空"})
  @IsString({message: "字符串类型"})
  @Length(2,10, {message: "2-10个字符"})
  @ApiProperty({description: "颜色", example: "red", required: false})
  color: string;

  @Column({ type: 'timestamp', default: () => 'current_timestamp' })
  createAt: Timestamp;

  @Column({
    type: 'timestamp',
    onUpdate: 'current_timestamp',
    default: () => 'current_timestamp',
  })
  updateAt: Timestamp;
}

export class CatC {

  @IsNotEmpty({message: "名称不能为空"})
  @IsString({message: "字符串类型"})
  @Length(2,10, {message: "2-10个字符"})
  @ApiProperty({description: "名称", example: "kition", required: true, nullable: false})
  name: string

  @IsNotEmpty({message: "年龄不能为空"})
  @IsInt({message: "int类型"})
  @Min(0, {message: "最小0"})
  @Max(100, {message: "最大100"})
  @ApiProperty({description: "年龄", example: 20, required: true, nullable: false})
  age: number

  color?: string
}

import { Module, MiddlewareConsumer, RequestMethod } from '@nestjs/common';
import { AppController } from './app.controller';
import { AppService } from './app.service';
import { UserModule } from './moduls/user/user.module';
import { AboutModule } from './moduls/about/about.module';

import { Log } from "./midway/log"
import { FileModule } from './moduls/file/file.module';
import { TaskModule } from './moduls/task/task.module';
import { ScheduleModule } from '@nestjs/schedule'

import { TypeOrmModule } from '@nestjs/typeorm'; // +
import { Cat } from "./db/mariadb/cat.entity" // +

@Module({
  imports: [UserModule, AboutModule, FileModule, TaskModule, 
    ScheduleModule.forRoot(),
    TypeOrmModule.forRoot({ // +
      type: 'mariadb',
      host: '192.168.2.200',
      port: 3306,
      username: 'root',
      password: '123456',
      database: 'test',
      entities: [Cat],
      synchronize: true,
    })
  ],
  controllers: [AppController],
  providers: [AppService],
})
export class AppModule {
  configure(consumer: MiddlewareConsumer) {
    consumer
      .apply(Log)
      .exclude({path: "/user/list", method: RequestMethod.GET}) // 忽略路径
      .forRoutes("/") // 中间件根路由
  }
}

        this.redis = this.redisServer.getClient("dev")
      }
    getUser(id: number): User{
        return users.find(ele => ele.id === id);
    }

    async createUser(user: UserC): User {
        await this.redis.set("hello", "world")
        const u = {
            ...user,
            id: users.length + 1,
            disable: false,
            achievement: [],
            parents: {
                father: "",
                mother: ""
            }
        };
        users.push(u);
        return u;
    }

    updateUser(user: User): User {
        const u = users.find(ele => ele.id === user.id)
        if(!u) {
            return;
        }
        Object.keys(user).forEach(key => {
            u[key] = user[key];
        });
        return u;
    }

    getUserList(): User[] {
        return users;
    }
}

import { Controller, Post, UseInterceptors ,UploadedFile, UploadedFiles, Body, Get, Res} from '@nestjs/common';
import { FileInterceptor, FilesInterceptor, FileFieldsInterceptor } from '@nestjs/platform-express';
import { writeFileSync } from 'fs'
import { join } from "path"
@Controller('file')
export class FileController {

    // 上传单个文件
    @Post()
    // "file" 表示 上传文件的键名
    @UseInterceptors(FileInterceptor("file"))
    upload(@UploadedFile() file, @Body() body) {
        writeFileSync(join(process.cwd(), `src/moduls/file/file/${file.originalname}`), file.buffer)
    }

    // 上传多个文件
    @Post('many')
    // "files" 表示 上传文件的键名
    @UseInterceptors(FilesInterceptor("files"))
    uploadMultipleKey(@UploadedFiles() files, @Body() body) {
        files.forEach(file => writeFileSync(join(process.cwd(), `src/moduls/file/file/${file.originalname}`), file.buffer))
    }

    // 上传多个文件(限制最大个数)
    @Post('limitNum')
    @UseInterceptors(FileFieldsInterceptor([
        { name: "files", maxCount:3 }, // maxCount文件个数,不写表示不限制
        { name: "custom" }
    ]))
    uploadMultipleKeyLimit(@UploadedFiles() files, @Body() body) {
        // 这里files 将被改为对象 {files:[File],custom:[File]}
        files.files.forEach(file => writeFileSync(join(process.cwd(), `src/moduls/file/file/${file.originalname}`), file.buffer))
    }
}

import { Injectable, Logger } from '@nestjs/common';
import { Cron, Interval, Timeout } from '@nestjs/schedule';

@Injectable()
export class TaskService {
  private readonly logger = new Logger(TaskService.name);

  @Cron('45 * * * * *')  
  handleCron() {
    this.logger.debug('Called when the second is 45');
  }

  @Cron('18 21 * * *') // 每天21:18执行
  handleCron2() {
    this.logger.debug('18---------------------------');
  }

  @Interval(10000) 
  handleInterval() {
    this.logger.debug('Called every 10 seconds');
  }

  @Timeout(5000) 
  handleTimeout() {
    this.logger.debug('Called once after 5 seconds');
  }
}
