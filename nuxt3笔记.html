# 动态路由

创建由[].vue的文件即可表示动态路由 例如:

vim pages/demo/[name].vue

<template>
    <div>
        name: {{ route.params.name }}
    </div>
</template>
<script setup lang="ts">
import { useRoute } from 'vue-router'
const route = useRoute()
</script>
<style scoped></style>

若同一路由下既有普通路由，也有动态路由，优先解析普通路由 比如访问http://localhost:8000/demo/c1 访解析到 pages/c1.vue,访问http://localhost:8000/demo/c11解析到 pages/[name].vue

<template>

# 请求

创建server/api文件夹，下面的每个文件会被解析成接口，前缀是/api，若不想要前缀则创建routes文件夹

routes和api的区别是routes没有前缀

根据文件名后缀可以设置请求方式，不写默认允许所有方式

例如server/api/todo.ts表示可接受所有请求

例如server/api/todo.get.ts表示只接受get请求

例如server/api/todo.post.ts表示只接受post请求

vim server/api/todo.ts

export const resData = {
    name: '',
    list: ['a', 'b', 'c'].map(val => ({val}))
}
export default defineEventHandler(async (h) => {
    await new Promise(t => {
        const timmer = setTimeout(() => {
            clearTimeout(timmer)
            t(1)
        }, 2000)
    })
    resData.name = 'get'
    return resData
})

vim pages/fetch.vue

<template>
    <div>
        Fetch
        <h1>server data: {{data?.name}}</h1>
        <p v-for="item in data?.list" :key="item.val">{{item.val}}</p>
    </div>
</template>

<script setup lang="ts">
import { resData } from '../server/api/todo2'
const {data} = await useFetch<typeof resData>('/api/todo', {
    method: 'GET',
    headers: {
        name: 'kition'
    },
    server: true, // 是否在服务器完成请求，默认true
    lazy: false, // 是否在路由之后请求数据(卡住路由),默认false，会卡住
    transform(input) { // 使用transform对返回对象进行定制化操作
        input.name += '经过transform转换'
        return input
    }
})
</script>

<style lang="scss" scoped></style>

export default defineEventHandler(async (h) => {
    const { params } = h.context
    const body = await readBody(h)
    const query = getQuery(h)
    const myHeader = getHeader(h, 'myHeader')
    console.log('input')
    console.log(params)
    console.log(query)
    console.log(body)
    console.log(myHeader)
    return 'ok'
})

前端

<template>

# nuxt3代理DELETE请求失败

nuxt3在以下两种情况下都不能代理delete请求 1.通过配置nuxt.config.ts下nitro.routeRules 2.通过server/apis/[...].ts(使用useFetch/$fetch)代理 可能是useFetch/$fetch的问题，换成axios可以解决

# 处理办法

配置nuxt.config.ts下nitro.routeRules来代理虽然简单，但是不灵活。比如这个问题就没法处理

还是通过server/apis/[...].ts(使用axios)代理:

pnpm add axios
mkdir -p server/apis
vim server/apis/[...].ts

import axios, { AxiosHeaders } from 'axios'
const runtimeConfig = useRuntimeConfig()
export const withBodyMethods = ['POST', 'PUT', 'Patch']

export default defineEventHandler(async (event) => {
    const method = getMethod(event)
    const res = await axios({
        url: `${runtimeConfig.proxyHost}${getRequestPath(event)}`,
        method,
        params: getQuery(event),
        data: withBodyMethods.includes(method.toUpperCase()) ? await readBody(event) : void 0,
        headers: getHeaders(event) as unknown as AxiosHeaders 
    }).then(res => res)
    .catch(err => err)
    if (axios.isAxiosError(res)) {
        throw createError({
            statusCode: res.response?.status,
            message: res.response?.data?.msg || '',
        })
    }
    return res.data
})
