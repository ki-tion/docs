pdf.js+带线条标注

最近公司有个功能，要求渲染pdf并允许标注。效果大致如下:

# 思路

选用pdf.js渲染pdf文件，通过getSelection获取用户选取文本，使用range获取文本的位置。注意需要使用getClientRects,若使用getBoundingClientRect可能会导致选中文字时，选中的文字会超出标注框。将一个canvas放置在渲染的pdf之上，这里需要注意canvas要去掉用户事件，否则pdf会被挡住而无法选中，将刚才的标注位置渲染到canvas上，然后通过标注位置生成对应的标签位置。至于两点之间的连线使用贝塞尔曲线连接起来。

                            const { height } = viewer.getBoundingClientRect();
                            this.content.style.height = height + "px";
                            this.canvas.height = height;
                            this.canvas.width = wrapper.scrollWidth;
                            const { height: viewerContainerHeight } =
                                viewer.getBoundingClientRect();
                            this.iframeWin.PDFViewerApplication.pdfViewer.currentScaleValue =
                                "page-actual"; // 实际大小，可缩放的实现起来太麻烦
                            this.firstPainting();
                        }, 50);
                        const observer = new MutationObserver((mutations) => {
                            after();
                        });
                        observer.observe(viewer, {
                            attributes: true,
                            childList: true,
                            subtree: true,
                        });
                        reslove(true);
                    };
                    this.iframe.onerror = () => {
                        reject(false);
                    };
                });
            }

            drawLine(label, rect, scrollTop, pageOffserXDiff) {
                const { x, y } = label.getBoundingClientRect();
                const startPoint = { x: rect.x + pageOffserXDiff, y: rect.y }
                const endPoint = { x: x, y: y + scrollTop }
                const xLen = endPoint.x - startPoint.x
                const yLen = endPoint.y - startPoint.y
                const b1Point = { x: startPoint.x + xLen * 0.25, y: startPoint.y  }
                const b2Point = { x: endPoint.x - xLen * 0.25, y: endPoint.y}

                this.ctx.globalAlpha = 0.5;
                this.ctx.lineWidth = 5;
                this.ctx.beginPath();
                this.ctx.moveTo(startPoint.x + rect.width, startPoint.y + 3); // 3 修正视觉偏差
                this.ctx.bezierCurveTo(b1Point.x, b1Point.y, b2Point.x, b2Point.y, endPoint.x, endPoint.y);
                this.ctx.strokeStyle = "red";
                this.ctx.stroke();
            }

                if (!page) {

                const pageOffserXDiff = pRect.x - mark.pageOffectX;

            }

            useLableEvent(label) {
                let x = 0;
                let y = 0;

                const ondrag =  debounce((event) => {
                    let prevX;
                        let prevY;
                        const transform = window
                            .getComputedStyle(label)
                            .getPropertyValue("transform");

                        // 如果元素已经移动过，获取元素当前的位置
                        if (transform && transform !== "none") {
                            let matrix = transform.replace(/[^0-9\-.,]/g, "").split(",");
                            prevX = parseFloat(matrix[12] || matrix[4]);
                            prevY = parseFloat(matrix[13] || matrix[5]);
                        } else {
                            prevX = 0;
                            prevY = 0;
                        }
                        const deltaX = event.clientX - x;
                        const deltaY = event.clientY - y;

                        const lsRect = this.labels.getBoundingClientRect()
                        const lRect = label.getBoundingClientRect()
                        if(lRect.left+deltaX<lsRect.left || lRect.right+deltaX>lsRect.right) { // 超出显示范围
                            label.style.border = "1px solid red"
                        } else {
                            label.style.border = "none"
                        }



                }, 50)
                return {
                    onDragStart: (event) => {
                        event.dataTransfer.setData("text/plain", event.target.id);
                        x = event.clientX;
                        y = event.clientY;
                    },
                    ondrag,
                    onDragEnd: (event) => {
                        let prevX;
                        let prevY;
                        const transform = window
                            .getComputedStyle(label)
                            .getPropertyValue("transform");

                        // 如果元素已经移动过，获取元素当前的位置
                        if (transform && transform !== "none") {
                            let matrix = transform.replace(/[^0-9\-.,]/g, "").split(",");
                            prevX = parseFloat(matrix[12] || matrix[4]);
                            prevY = parseFloat(matrix[13] || matrix[5]);
                        } else {
                            prevX = 0;
                            prevY = 0;
                        }
                        const deltaX = event.clientX - x;
                        const deltaY = event.clientY - y;

                        const lsRect = this.labels.getBoundingClientRect()
                        const lRect = label.getBoundingClientRect()
                        label.style.border = "none"
                        // if(lRect.left+deltaX<lsRect.left - lRect.width * 0.5 || lRect.right+deltaX>lsRect.right + lRect.width * 0.5) { // 超出显示范围
                        if(lRect.left+deltaX<lsRect.left || lRect.right+deltaX>lsRect.right) { // 超出显示范围
                            return
                        }
                        
                        label.style.transform = `translate3d(${prevX + deltaX}px, ${prevY + deltaY
                            }px, 0)`;
                        this.redraw();
                    },
                    onClick: (event) => {
                        if (event.target.className === 'layui-icon layui-icon-edit') {
                            this.labelUpdateIconClick(label)
                        } else if (event.target.className === 'layui-icon layui-icon-delete') {
                            this.labelDeleteIconClick(label)
                        } else {
                            console.log('暂无操作')
                        }
                    }
                };
            }
