# 路由

添加路由出口NuxtPage

vim app.vue

<template>
  <div>
    <!-- 删除默认页面 -->
    <!-- <NuxtWelcome /> -->
    <NuxtPage></NuxtPage>
  </div>
</template>

约定/pages下文件(夹)为路由。默认没有创建该文件夹

mkdir /pages

vim /pages/index.vue

index翻译成'/'

<template>
    <div>
        <h1>{{name}}</h1>
    </div>
</template>

<script setup lang="ts">
const name = "index page"
</script>

# 嵌套路由

创建一个和页面同名的文件夹，该文件夹里面的文件会被解析成子路由。

vim /pages/father.vue

创建父路由,记得加上子路由出口NuxtChild

<template>
    <div>
        <h1>{{name}}</h1>
        <NuxtChild></NuxtChild>
    </div>
</template>

<script setup lang="ts">
const name = "father page"
</script>

创建子路由

mkdir /pages/father
vim /pages/father/child.vue

<template>
    <div>
        <h1>{{name}}</h1>
    </div>
</template>

<script setup lang="ts">
const name = "child page"
</script>

# 动态路由

文件或文件夹放入[]，或被解析成parm

vim /pages/parm-[id].vue

<template>
    <div>
        <h1>{{name}}</h1>
        <p>{{ $route.params.id }}</p>
    </div>
</template>

<script setup lang="ts">
const name = "parm page"
</script>

访问: http://localhost:3000/parm-1223

mkdir /pages/parm2
vim /pages/parm2/[name].vue

<template>
    <div>
        <h1>{{name}}</h1>
        <p>{{ $route.params.name }}</p>
    </div>
</template>

<script setup lang="ts">
const name = "parm2 page"
</script>

访问: http://localhost:3000/parm2/aaaa

# 404页面

# layout

/layouts文件夹下的文件会被加载成布局文件

mkdir /layouts
vim /layouts/main.vue

<template>
    <div>
        <h1>{{name}}</h1>
        <!-- 注意添加插槽,layout依赖插槽实现 -->
        <slot></slot>
    </div>
</template>

<script setup lang="ts">
const name = "main layout"
</script>

vim /pages/index.vue

<template>
    <!-- name选取layout -->
    <NuxtLayout name="main">
        <h1>{{name}}</h1>
        <!-- 不需要手动注册 -->
        <compA></compA>
        <BaseFoo></BaseFoo>
        <!-- 加上Lazy即是懒加载组件 -->
        <LazyBaseFoo></LazyBaseFoo>
    </NuxtLayout>
</template>

<!-- 注意:两个script lang需要一致 -->
<script lang="ts">
export default {
  // 使用NuxtLayout标签选择layout，可以关闭这个
  layout: false,
}
</script>
<script setup lang="ts">
const name = "index page"
</script>

# 组合逻辑

组合逻辑可以共享状态，理解成vuex就好。组合逻辑基于useState实现。

/composables目录下创建文件会被处理成组合逻辑

mkdir /composables

# 运行时配置

运行时配置分为公开的和私有的，都在 nuxt.config 文件中定义

client端只可以使用公开配置

vim /nuxt.config.ts

import { defineNuxtConfig } from 'nuxt3'

// https://v3.nuxtjs.org/docs/directory-structure/nuxt.config
export default defineNuxtConfig({
    // 公开
    publicRuntimeConfig: {
        API_BASE: "/api"
    },
    // 私有
    privateRuntimeConfig: {
        API_SECRET: "123"
    }
})

// 通过#config导入
import config from '#config'
export default async (req, res) => {
    await (() => new Promise(t => setTimeout(() => t(true), 2000)))()
    res.statusCode = 200
    res.end(JSON.stringify({
        errNo: 0,
        list: [
            {
                name: req.Token,
                age: 20

        ]
    }))
}
